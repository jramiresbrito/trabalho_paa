def display(assets_list)
  assets_list.each do |asset|
    msg = "#{'Ativo:'.red} #{asset[:asset]} "
    msg += "#{'Retorno Esperado:'.blue} #{adjust_leading_zeros(asset[:expected_return] * 100)}%\n"

    print msg
  end
